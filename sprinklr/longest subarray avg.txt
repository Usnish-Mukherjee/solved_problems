#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> ip;
static bool cmp(ip &a,ip &b)
{
    return (a.first>b.first) || ((a.first==b.first)&&(a.second<b.second));
}
int util(vector<ip> &v,int n,int k)
{
    int start=0,end=n-1;
    int ans=-1;
    while(start<=end)
    {
        int mid=start+(end-start)/2;
        if(v[mid].first>=k)
        {
            ans=mid;
            start=mid+1;
        }
        else
            end=mid-1;
    }
    return ans;
}
int main()
{
    int n,k;
    cin>>n>>k;
    int a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<n;i++)
        a[i]=a[i]-k;
    k=0;
    int sum=0;
    vector<ip> v;
    for(int i=0;i<n;i++)
    {
        sum+=a[i];
        v.push_back({sum,i});
    }
    sort(v.begin(),v.end(),cmp);
    int mini[n];
    mini[0]=v[0].second;
    for(int i=1;i<n;i++)
    {
        mini[i]=min(mini[i-1],v[i].second);
    }
    sum=0;
    int maxlen=0;
    for(int i=0;i<n;i++)
    {
        sum+=a[i];
        if(sum<=k)
            maxlen=max(maxlen,i+1);
        else
        {
            int id=util(v,n,sum-k);
            if(id!=-1 && mini[id]<i)
                maxlen=max(maxlen,i-mini[id]);
        }
    }
   cout<<maxlen;
}
