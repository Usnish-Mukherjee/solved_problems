#include<bits/stdc++.h>
using namespace std;
bool manachar(string s)
{
  string t;
  t+="$";
  for(int i=0;i<s.size();i++)
  {
    t+="#";
    t+=s[i];
  }
  t+="#@";
  int c=0,r=0;
  vector<int> p(t.size(),0);
  for(int i=1;i<t.size();i++)
  {
    int mirr=2*c-i;
    if(i<r)
    p[i]=min(r-i,p[mirr]);
    while(t[i+(p[i]+1)]==t[i-(p[i]+1)])
    p[i]++;
    if(r<i+p[i])
    {
      r=i+p[i];
      c=i;
    }
  }
  int flag=0;
  int start=1,end=p.size()-2;
  for(int i=2;i<p.size()-2;i+=2)
  {
    if(i-start<=end-i)
    {
      if(i-p[i]>start)
        return false;
    }
    else
    {
      if(i+p[i]<end)
        return false;
    }
  }
  return true;
}
int main()
{
  string s;
  cin>>s;
   cout<<  manachar(s);
  return 0;
}
